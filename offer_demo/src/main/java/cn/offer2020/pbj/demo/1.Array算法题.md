#### 26. 删除排序数组中的重复项
```markdown
public int removeDuplicates(int[] nums) {
        if(nums==null||nums.length==0) return 0;
        int i = 0;
        int j = 1;
        while(j < nums.length){
            if(nums[i]!=nums[j]){
                nums[i+1]= nums[j];
                i++;
            }
            j++;
        }
        return i+1;
    }
```

```markdown
        }
```

```markdown
        int j = 0;
        int n = nums.length;
        while(j<n){
            if(nums[j]==val){
                nums[j] = nums[n-1];
                n--;
            }else{
                j++;
            }
        }
        return n;
    }
```

```markdown
public void merge(int[] nums1, int m, int[] nums2, int n) {
        int l = m + n - 1;
        int i = m - 1;
        int j = n - 1;
        while(i>=0&&j>=0){
            if(nums1[i]<nums2[j]){
                nums1[l--] = nums2[j--];
            }else{
                nums1[l--] = nums1[i--];
            }
        }
        while(j>=0){
            nums1[l--] = nums2[j--];
        }
    }
```

```markdown
public int majorityElement(int[] nums) {
        int count = 1;
        int res = nums[0];
        for(int i = 1;i<nums.length;++i){
            if(count==0){
                res = nums[i];
                count = 1;
            }else if(nums[i] == res){
                count++;
            }else{
                count--;
            }
        }
        return res;
    }
```

#### 229. 求众数 II
```markdown
public List<Integer> majorityElement(int[] nums) {
        List<Integer> ans = new ArrayList<>();
        int count1 = 0, temp1 = 0;
        int count2 = 0, temp2 = 0;
        for (int i = 0; i < nums.length; i++) {
            if ((count1 == 0 || temp1 == nums[i]) && temp2 != nums[i]) {
                count1++;
                temp1 = nums[i];
            } else if (count2 == 0 || temp2 == nums[i]) {
                count2++;
                temp2 = nums[i];
            } else {
                count1--;
                count2--;
            }
        }
        count1 = 0;
        count2 = 0;
        for(int i = 0;i < nums.length;i++) {
            if(nums[i] == temp1)
                count1++;
            else if(nums[i] == temp2)
                count2++;
        }
        if(count1 > nums.length / 3)
            ans.add(temp1);
        if(temp1 != temp2 && count2 > nums.length / 3)
            ans.add(temp2);
        return ans;
    }
```

#### 238. 除自身以外数组的乘积

```markdown
public int[] productExceptSelf(int[] nums) {
        int len = nums.length;
        int[] res = new int[len];
        for(int i = 0,curr = 1;i<len;i++){
            res[i] = curr;
            curr = curr * nums[i];
        }
        for(int i = len-1,curr = 1;i>=0;i--){
            res[i] = res[i]*curr;
            curr = curr * nums[i];
        }
        return res;
    }
```

#### 54. 螺旋矩阵
```markdown
public List<Integer> spiralOrder(int[][] matrix) {
        if(matrix==null||matrix.length==0||matrix[0].length==0){
            return new LinkedList<>();
        }
        int l = 0;
        int r = matrix[0].length-1;
        int u = 0;
        int d = matrix.length-1;
        List<Integer> list = new LinkedList<>();
        while(l<=r&&u<=d){
            for(int i = l;i<=r;i++){
                list.add(matrix[u][i]);
            }
            u++;
            for(int i = u;i<=d;i++){
                list.add(matrix[i][r]);
            }
            r--;
            for(int i = r;i>=l&&u<=d;i--){
                list.add(matrix[d][i]);
            }
            d--;
            for(int i = d;i>=u&&l<=r;i--){
                list.add(matrix[i][l]);
            }
            l++;
        }
        return list;
    }
```

```markdown
//生成一个包含 1 到 n2 所有元素，且元素按顺时针顺序螺旋排列的正方形矩阵。
public int[][] generateMatrix(int n) {
        int[][] ans = new int[n][n];
        int c = 1,j = 0;
        while(c<=n*n){
            for(int i = j;i<n-j;i++){
                ans[j][i] = c++;
            }
            for(int i = j+1;i<n-j;i++){
                ans[i][n-j-1]= c++;
            }
            for(int i = n-j-2;i>=j;i--){
                ans[n-j-1][i] = c++;
            }
            for(int i = n-j-2;i>j;i--){
                ans[i][j] = c++;
            }
            j++;
        }
        return ans;
    }
```

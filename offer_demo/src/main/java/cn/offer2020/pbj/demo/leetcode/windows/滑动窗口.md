public class Main{

    int left = 0,right = 0;
    String res = s;
    while(right < s.length()){
        window.add(s[right]);
        right++;
        // 如果符合要求，扫描窗口构造完成，移动left缩小窗口
        while(window 符合要求){
            //如果这个窗口的子串更短，则更新res
            res = minLen(res,window);
            window.remove(s[left]);
            left++;
        }
    }
    return res;
}

```java
public class Demo2016 {
    // 通过数学分析，最后获得的饮料数是总空瓶数整除2 。
    public static void main(String[] args) {
        List<Integer> inputs = new ArrayList<>();
        Scanner sc = new Scanner(System.in);
        while (true) {
            int input = sc.nextInt();
            if (0 == input) {
                break;
            } else {
                inputs.add(input);
            }
        }
        sc.close();
        for (int input : inputs) {
            int number = help(input / 3, input - 3 * (input / 3));
            System.out.println(number);
        }
    }

    public static int help(int drunk, int bottleNumber) {
        int total = drunk + bottleNumber;
        if (total > 2) {
            return drunk + help(total / 3, total - 3 * (total / 3));
        } else if (total == 2) {
            return drunk + 1;
        } else {
            return drunk;
        }
    }
}
```

```java
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        while (scanner.hasNext()) {
            int n = scanner.nextInt();
            int[] intArr = new int[n];
            for (int i = 0; i < n; i++) {
                intArr[i] = scanner.nextInt();
            }
            Arrays.sort(intArr);
            for (int i = 0; i < intArr.length; i++) {
                // 第一个数字或者不等于前一个数字都可以输出
                if (i == 0 || intArr[i] != intArr[i - 1]) {
                    System.out.println(intArr[i]);
                }
            }
        }
    }
}
```

### 第二题 走路
```java
/**
* 
  小明步长 l, 有一个 m x n 的矩阵，1表示能走0表示不能走，小明可以横着走竖着走，方向不限制，问小明能否从左上角走到右下角。
  输入：
  2
  3 5
  1 0 1 0 0
  0 1 1 0 1
  0 0 1 0 1`
  输出：1
  简单的 DFS，在此不表。
    动态规划，从终点往前，看每个格子能否到终点，如果一个格子能到一个能到终点的格子，那他就可以到终点
*/
public class Main{
    public static void main(String[] args){
        Scanner sc  = new Scanner(System.in);
        int size = sc.nextInt();
        int m = sc.nextInt();
        int n = sc.nextInt();
        int[][] nums = new int[m][n];
        boolean[][] visited = new boolean[m][n]
        for(int i = 0;i<m;i++){
            for(int j = 0;j<n;j++){
                nums[i][j] = sc.nextInt();
                visited[i][j] = false;
            }
        }
        boolean res = dfs(nums,size,0,0,visited);
        if(res){
            System.out.println(1);
        }else {
            System.out.println(0);
        }
    }

    private boolean dfs(int[][] nums,int size,int i,int j,boolean[][] visited){
        if(i<0||j<0||i>=nums.length||j>=nums[0].length||nums[i][j]==0) return false;
        if(i==nums.length-1&&j==nums[0].length-1) return true;
        boolean ans = dfs(nums,size,i-size,j)||dfs(nums,size,i+size,j)||dfs(nums,size,i,j-size)||dfs(nums,size,i,j+size);
    }
}
```
